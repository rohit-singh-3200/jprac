
from sklearn.datasets import make_blobs
from sklearn.cluster import KMeans, AgglomerativeClustering
from sklearn.metrics import silhouette_score
import matplotlib.pyplot as plt
import numpy as np

X, y_true = make_blobs(n_samples=300, centers=3, cluster_std=1.0, random_state=42)

# KMeans
kmeans = KMeans(n_clusters=3, random_state=42)
labels_k = kmeans.fit_predict(X)
print('KMeans inertia:', kmeans.inertia_)
print('KMeans silhouette:', silhouette_score(X, labels_k))

# Agglomerative
agg = AgglomerativeClustering(n_clusters=3)
labels_a = agg.fit_predict(X)
print('Agglomerative silhouette:', silhouette_score(X, labels_a))

# Plot results side by side
plt.figure(figsize=(12,5))
plt.subplot(1,2,1)
plt.title('KMeans')
plt.scatter(X[:,0], X[:,1], c=labels_k, s=10)
plt.subplot(1,2,2)
plt.title('Agglomerative')
plt.scatter(X[:,0], X[:,1], c=labels_a, s=10)
plt.show()
